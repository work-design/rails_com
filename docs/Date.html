def after(other)
  if ActiveSupport::Duration === other
    if other.parts.keys == [:months]
      date = self + other
      if date.day == self.day
        r = date - 1.day
      else
        r = date + (self.day - date.day - 1).days
      end

      r
    else
      self + other
    end
  else
    self + other
  end
end
